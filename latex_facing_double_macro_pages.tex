\newcommand\FDB[4]{
\begin{textblock}{10}(0,#1)
%\textblockcolour{cyan!10} % ugly, only for testing
\section{#2\label{#2}} #3 \bigskip 
\end{textblock}
\afterpage{
\begin{textblock}{10}(0,#1)
% \textblockcolour{green!10} % ugly, only for testing
\section*{Annonations about section \ref{#2}}
 #4\bigskip
\end{textblock}}}

\textblockcolour{green!10} % ugly, only for testing
#3 \smallskip
\end{textblock}}}
